-- Создание базы данных
CREATE DATABASE IncidentManagmentSystem;
GO

USE IncidentManagmentSystem;
GO

-- 1. Таблица сотрудников (Employe)
CREATE TABLE Employees (
    Id INT IDENTITY(1,1) PRIMARY KEY,
    Name NVARCHAR(200) NOT NULL,
    Email NVARCHAR(200) NOT NULL UNIQUE,
    Login NVARCHAR(100) NOT NULL UNIQUE,
    Password NVARCHAR(255) NOT NULL,
    CreatedAt DATETIME DEFAULT GETDATE()
);
GO

-- 2. Таблица категорий (Category)
CREATE TABLE Categories (
    Id INT IDENTITY(1,1) PRIMARY KEY,
    Name NVARCHAR(100) NOT NULL UNIQUE
);
GO

-- 3. Таблица подразделений (Subdivision)
CREATE TABLE Subdivisions (
    Id INT IDENTITY(1,1) PRIMARY KEY,
    Name NVARCHAR(200) NOT NULL UNIQUE
);
GO

-- 4. Таблица статусов (Status)
CREATE TABLE Statuses (
    Id INT IDENTITY(1,1) PRIMARY KEY,
    Name NVARCHAR(50) NOT NULL UNIQUE
);
GO

-- 5. Основная таблица инцидентов (Incident)
CREATE TABLE Incidents (
    Id INT IDENTITY(1,1) PRIMARY KEY,
    AuthorName NVARCHAR(200) NOT NULL,
    AuthorEmail NVARCHAR(200) NOT NULL,
    Description NVARCHAR(MAX) NOT NULL,
    Priority INT NOT NULL CHECK (Priority BETWEEN 1 AND 5),  -- 1 - наивысший, 5 - низший
    CategoryId INT NOT NULL,
    SubdivisionId INT NOT NULL,
    EmployeId INT NOT NULL,
    StatusId INT NOT NULL,
    CreateDate DATETIME NOT NULL DEFAULT GETDATE(),
    FinalDate DATETIME NOT NULL ,  -- может быть NULL, пока инцидент не закрыт
    
    -- Внешние ключи
    CONSTRAINT FK_Incident_Category FOREIGN KEY (CategoryId) REFERENCES Categories(Id),
    CONSTRAINT FK_Incident_Subdivision FOREIGN KEY (SubdivisionId) REFERENCES Subdivisions(Id),
    CONSTRAINT FK_Incident_Employe FOREIGN KEY (EmployeId) REFERENCES Employees(Id),
    CONSTRAINT FK_Incident_Status FOREIGN KEY (StatusId) REFERENCES Statuses(Id)
);
GO

-- 6. Таблица файлов (File) - привязана к инцидентам
CREATE TABLE Files (
    Id INT IDENTITY(1,1) PRIMARY KEY,
    IncidentId INT NOT NULL,
    FileName NVARCHAR(255) NOT NULL,
    ContentType NVARCHAR(100) NOT NULL,
    FileData VARBINARY(MAX) NOT NULL,
    FileSize BIGINT NOT NULL,
    UploadDate DATETIME DEFAULT GETDATE(),
    
    CONSTRAINT FK_File_Incident FOREIGN KEY (IncidentId) REFERENCES Incidents(Id) ON DELETE CASCADE
);
GO

-- 7. Таблица истории изменений (History)
CREATE TABLE Histories (
    Id INT IDENTITY(1,1) PRIMARY KEY,
    IncidentId INT NOT NULL,
    EmployeId INT NOT NULL,
    Description NVARCHAR(MAX) NOT NULL,
    Zametka NVARCHAR(MAX) NULL,
    ChangeDate DATETIME DEFAULT GETDATE(),
    
    CONSTRAINT FK_History_Incident FOREIGN KEY (IncidentId) REFERENCES Incidents(Id) ON DELETE CASCADE,
    CONSTRAINT FK_History_Employe FOREIGN KEY (EmployeId) REFERENCES Employees(Id)
);
GO
